def print_Confusion_Matrix(t,y):
  print(confusion_matrix(t,y))

def score(t:torch.Tensor,y:torch.Tensor):
    label_names = [f"Label: {_}" for _ in range(10)]
    return classification_report(t,y,target_names=label_names)

def get_balanced_accuracy(y_pred,y_true):
    from sklearn.metrics import balanced_accuracy_score
    return balanced_accuracy_score(y_true, y_pred)

def predict(model,data):
  y_pred = []
  y_true = []
  for i,(images,labels) in enumerate(data):
    _ , pred = torch.max(model(images),axis=1)
    y_pred += list(pred.numpy())
    y_true += list(labels.numpy())
  return np.array(y_pred),np.array(y_true)

def scoresForAll(model: nn.Sequential):
  for key,loader in data_loaders.items(): 
    print(f'\n\nFor : {key}\n')
    y , t = predict(model,loader)
    print(score(t,y))
    b = get_balanced_accuracy(y,t)
    print(f'\n\nbalanced accuracy is : {b}\n\n')


def accuracy(y, t):
  score, predicted = torch.max(y, 1)
  acc = (t == predicted).sum().float() / len(t)
  return acc

def train(model: nn.Sequential,optimizer,forCheck = None,loader = None):
  import time
  if loader == None:
    loader = train_loader
  if (forCheck == None) or (forCheck == False):
    model.to(device)
    start = time.time()
    for epoch in range(num_epochs):
      for images, labels in loader:
        cross_entropy = loss(model(images), labels)
        optimizer.zero_grad()
        cross_entropy.backward()
        optimizer.step()

    end = time.time()
    print(f'time : {end-start}\nsummary : \n')
    summary(model,(1,28,28))
    scoresForAll(model)
    print("Final train CE: ", cross_entropy.cpu().detach().numpy(),'\n')
    print(f'Num of weights and biases : {sum(parameter.numel() for parameter in model.cpu().parameters())}')
    return
  temp = True
  iterations = 0
  model.to(device)
  start_time = time.time()  
  while temp:
    for images, labels in loader:
      last_acc = accuracy(model(images), labels).item()
      if last_acc >= 0.99:
        temp = False
      cross_entropy = loss(model(images), labels)
      optimizer.zero_grad()
      cross_entropy.backward()
      optimizer.step()
      iterations += 1
  end_time = time.time()
  summary(model,(1,28,28))
  print(f'time : {end_time-start_time}\nsummary : \n')
  scoresForAll(model)
  print(f'weight_num : {sum(param.numel() for param in model.cpu().parameters())}\nnum of iterations : {iterations}')

def init_conv2d_weights(m):
    if isinstance(m, nn.Conv2d):
        nn.init.xavier_normal_(m.weight)
        m.weight = nn.Parameter(torch.abs(m.weight))
        m.bias.data.fill_(0.01)

activation = {}
def get_activation(name):
    def hook(model, input, output):
        activation[name] = output.detach()
    return hook

def conv_image(model, batch, image_num, layer_idx, channel_idx, after_activation, axes_index):
  check_activation = "after activation = True" if after_activation else "after activation = False"
  layer_name = f"layer {layer_idx+1} {check_activation}"
  
  idx = layer_idx + 1 if after_activation else layer_idx
  model[idx].register_forward_hook(get_activation(layer_name))

  output = model(batch[0])
  axes_index.set_title(layer_name, fontSize=35)
  axes_index.imshow(activation[layer_name][image_num][channel_idx].cpu())
